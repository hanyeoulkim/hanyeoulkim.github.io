import pygame
import math
import random
 
pygame.init()
#껍데기 그려주는 역할
 
width, height = 640, 480
clock = pygame.time.Clock()
screen = pygame.display.set_mode((width, height))
#screen 정의 해주기.
playerpos = [100,100]
keys = dict()
keys['w'] = False
keys['a'] = False
keys['s'] = False
keys['d'] = False
#총알 관리해줄 배열 만들자.
bullets = []
badguys = []
badtimer = 100
#타이머가 줄어들면서 일정 숫자에 도달하면 배드가이가 출몰하도록 해보자.
 
player = pygame.image.load("../resources/images/dude.png")
grass = pygame.image.load("../resources/images/grass.png")
castle = pygame.image.load("../resources/images/castle.png")
bullet = pygame.image.load("../resources/images/bullet.png")
badguy = pygame.image.load("../resources/images/badguy.png")
 
while True:
 
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            pygame.quit()
        if event.type == pygame.KEYDOWN:
            if event.key == pygame.K_w:
                keys['w']=True
            if event.key == pygame.K_a:
                keys['a'] = True
            if event.key == pygame.K_s:
                keys['s'] = True
            if event.key == pygame.K_d:
                keys['d'] = True
        if event.type == pygame.KEYUP:
            if event.key == pygame.K_w:
                keys['w'] = False
            if event.key == pygame.K_a:
                keys['a'] = False
            if event.key == pygame.K_s:
                keys['s'] = False
            if event.key == pygame.K_d:
                keys['d'] = False
        if event.type == pygame.MOUSEBUTTONDOWN:
            m_pos = pygame.mouse.get_pos()
            b = [ math.atan2(m_pos[1] - playerpos[1], m_pos[0] - playerpos[0]),
                  playerpos[0], playerpos[1]]
            #쉼표 앞은 총 쏴질때 방향, 뒤 playerpos[0]과 [1]은 총알의 출발 좌표
            bullets.append( b )
 
    if keys['w']:
        playerpos[1] -= 20
        #player의 포지션 y 좌표에서 10을 빼겠다.
    if keys['a']:
        playerpos[0] -= 20
    if keys['s']:
        playerpos[1] += 20
    if keys['d']:
        playerpos[0] += 20
 
    #마우스 위치대로 돼지가 바라보도록 math 이용해서 조정해보자.
    m_pos = pygame.mouse.get_pos()
    angle = math.atan2(m_pos[1] - playerpos[1], m_pos[0] - playerpos[0])
    player_rotate = pygame.transform.rotate(player, 360-angle*57)
 
    for b in bullets:
        index = 0
        velx = math.cos(b[0])*10
        vely = math.sin(b[0])*10
        b[1] += velx
        b[2] += vely
 
        if b[1] < 0 or b[1] > 640 or b[2] <0 or b[2] > 480:
            bullets.pop(index)
            #배열에서 삭제해버리는 것.
        index += 1
 
    badtimer -= 1
    if badtimer == 0:
        badguys.append([640, random.randint(50, 430)])
        #640, 랜덤y 좌표에서 출몰한다.
        badtimer = random.randint(70, 100)
        #배드타이머는 70~100 사이의 숫자로 다시 바뀐다.
 
    index = 0
    for bg in badguys:
        if bg[0] < 0:
            badguys.pop(index)
        bg[0] -= 5
 
        badrect = pygame.Rect(badguy.get_rect())
        badrect.left = bg[0]
        badrect.top = bg[1]
        index1 = 0
        for b in bullets:
            b_rect = pygame.Rect(bullet.get_rect())
            b_rect.left = b[1]
            b_rect.top = b[2]
            #총알이 닿았는지 검사해주는 놈이다.
            if badrect.colliderect(b_rect):
                #colliderect는 배드가이의 사각형이 총알과 충돌했나 봐주는 놈이다.
                badguys.pop(index)
                bullets.pop(index1)
            index1 += 1
 
        index += 1
 
    screen.fill((0,0,0))
    for x in range(width // grass.get_width()+1):
        for y in range(height // grass.get_height()+1):
            screen.blit(grass, (x*100, y*100))
    screen.blit(castle, (0,30))
    screen.blit(castle, (0, 130))
    screen.blit(castle, (0, 240))
    screen.blit(castle, (0, 345))
 
#캐릭터가 움직일 수 있게 하기 위해 변수 처리 해보자.
    screen.blit(player_rotate, playerpos)
 
    for b2 in bullets:
        b2_rotate = pygame.transform.rotate(bullet, 360-b2[0]*57)
        screen.blit(b2_rotate,(b2[1], b2[2]))
    for bg in badguys:
        screen.blit(badguy,(bg[0], bg[1]))
 
    clock.tick(30)
    pygame.display.flip()
